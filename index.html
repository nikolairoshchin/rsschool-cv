    "twenty" => 20, "thirty" => 30,
    "forty" => 40, "fifty" => 50,
    "sixty" => 60, "seventy" => 70,
    "eighty" => 80, "ninety" => 90,
    "hundred" => 100, "thousand" => 1000,
    "million" => 1000000}
  m, n = 0
  summa = Array.new(0)
  string. gsub!(/-| and /, ' ')
  string. split. each do |x|
     
     n = numbers[x]
     case n
       when 100
         m *= n
       when 1000, 1000000
         summa << (m * n)
         m = 0
     
       else
         m += n
     end
   end
   
  return m + summa. sum
end
